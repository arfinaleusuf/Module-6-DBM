ANS: 1
    SELECT *
    FROM EMPLOYEES
    WHERE SALARY < (SELECT SALARY
			FROM EMPLOYEES
			WHERE EMPLOYEE_ID = 144); 

ANS:2
    SELECT *
    FROM EMPLOYEES
    WHERE SALARY = (SELECT MAX(SALARY)
				FROM EMPLOYEES);    --144 ER CHEYE KOM SALARY KE PAY

ANS: 3
    SELECT *
    FROM EMPLOYEES
    WHERE DEPARTMENT_ID = (SELECT DEPARTMENT_ID
    						FROM DEPARTMENTS
                            WHERE DEPARTMENT_NAME = 'MARKETING'
                            );

ANS: 4
    SELECT AVG(SALARY)
    FROM EMPLOYEES
    WHERE DEPARTMENT_ID = (SELECT DEPARTMENT_ID
	    					FROM DEPARTMENTS
                            WHERE DEPARTMENT_NAME = 'MARKETING'
                            );

ANS: 5
    SELECT COUNT(*) AS MEMBERS
    FROM EMPLOYEES
    WHERE DEPARTMENT_ID = (SELECT DEPARTMENT_ID
	    					FROM DEPARTMENTS
                            WHERE DEPARTMENT_NAME = 'IT'
                            );                            

ANS: 6
SELECT SUM(SALARY)
FROM EMPLOYEES
