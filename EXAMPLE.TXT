1. EMPLOYEE_ID ER MODDHE KE KE 144 ER CHEYE KOM SALARY PAY?
2. HIGHEST SALARY KE PAY?
3. MARKETING DEPT E KE KE KAJ KORE?

ANS: 1
    SELECT *
    FROM EMPLOYEES
    WHERE SALARY < (SELECT SALARY
			FROM EMPLOYEES
			WHERE EMPLOYEE_ID = 144); 

ANS:2
    SELECT *
    FROM EMPLOYEES
    WHERE SALARY = (SELECT MAX(SALARY)
				FROM EMPLOYEES);    --144 ER CHEYE KOM SALARY KE PAY

ANS: 3
    SELECT *
    FROM EMPLOYEES
    WHERE DEPARTMENT_ID = (SELECT DEPARTMENT_ID
    						FROM DEPARTMENTS
                            WHERE DEPARTMENT_NAME = 'MARKETING'
                            );
